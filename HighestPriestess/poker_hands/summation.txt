Name = Summation
Desc = {
    3 or more cards with one card's rank being the sum of all the other
    cards' ranks. (A, J, Q, K count as 1, 11, 12, and 13 respectively.)
}
Base Chips = 12
Base Mult = 3
Level Chips = 11
Level Mult = 2
Eval = {
    (* of *, * of *, * of *),
}
Math Identity = Sum
Author = GoldenLeaf
Example = 2 of Hearts, Ace of Spades, 3 of Hearts, 6 of Clubs
